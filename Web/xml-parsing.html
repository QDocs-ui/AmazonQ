  <test1>haha</test1>
</test>
            


            
Complex XML with Attributes:

            

<Order id="ORD001" date="2024-01-01">
  <Customer id="CUST001" name="John Doe">
    <Address>
      <Street>123 Main St</Street>
      <City>New York</City>
      <Country>USA</Country>
    </Address>
  </Customer>
  <Items>
    <Item code="LAPTOP001" qty="1" price="999.99">
      <Description>Gaming Laptop</Description>
    </Item>
    <Item code="MOUSE001" qty="2" price="25.50">
      <Description>Wireless Mouse</Description>
    </Item>
  </Items>
</Order>
            


            
SOAP Envelope:

            

<soap:Envelope xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/">
  <soap:Header>
    <auth:Authentication xmlns:auth="http://example.com/auth">
      <auth:Username>user123</auth:Username>
      <auth:Password>pass123</auth:Password>
    </auth:Authentication>
  </soap:Header>
  <soap:Body>
    <ws:GetCustomer xmlns:ws="http://example.com/webservice">
      <ws:CustomerId>CUST001</ws:CustomerId>
    </ws:GetCustomer>
  </soap:Body>
</soap:Envelope>
